## L2-049 鱼与熊掌（哈希表）

题目链接：https://pintia.cn/problem-sets/994805046380707840/exam/problems/type/7?problemSetProblemId=1781658570803388424&page=1

题目分析：本题还是一个比较简单的哈希表的题目，具体看解题代码吧！

```cpp
#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;//从1编号
    int m;//物品种类的总数，从1编号
    cin>>n>>m;
    vector<unordered_set<int>> arr(n);
   for(int i=0;i<n;i++){
       int k; cin>>k;
       for(int j=0;j<k;j++){
           int x; cin>>x;
           arr[i].insert(x);
       }
   }
    int q; cin>>q;
    for(int i=0;i<q;i++){
        int a,b;
        cin>>a>>b;
        int cnt=0;
        for(int j=0;j<arr.size();j++){
            if(arr[j].count(a) && arr[j].count(b)){
                cnt++;
            }
        }
        cout<<cnt<<endl;
    }
    return 0;
}

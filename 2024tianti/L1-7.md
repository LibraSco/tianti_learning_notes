## L1-103 整数的持续性

题目链接：https://pintia.cn/problem-sets/994805046380707840/exam/problems/type/7?problemSetProblemId=1781658570803388422&page=1

**题目分析：本题主要是做一些数学运算，看题目，发现(b-a)的范围其实比较小，所以这道题可以暴力求解，具体看解法代码吧！**

```cpp
#include<bits/stdc++.h>
using namespace std;
int calculate(long long num){
    int step=0;
    if(num<=9){
        return step;
    }
    while(num>=10){
        long long res=1;
        while(num>0){
            int a=num%10;
            res*=a;
            num/=10;
        }
        step++;
        num=res;
    }
    return step;
}
int main(){
    long long a,b;
    cin>>a>>b;
    int max_s=0;
    vector<int> result;
    for(long long i=a;i<=b;i++){
        int temp=calculate(i);
        if(temp>max_s){
            result.clear();
            result.push_back(i);
            max_s=temp;
        }
        else if(temp==max_s){
            result.push_back(i);
        }
    }
    cout<<max_s<<endl;
    for(int i=0;i<result.size();i++){
        if(i>0){
            cout<<" ";
        }
        cout<<result[i];
    }
    return 0;
}

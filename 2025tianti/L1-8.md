## L1-112 现代战争

题目链接：https://pintia.cn/problem-sets/994805046380707840/exam/problems/type/7?problemSetProblemId=1913922872972247047&page=1

**题目分析：这道题我想的很简单，就是先创建一个矩阵，每次遍历矩阵，求出最大价值的点，每次投下炸弹，
就将对应的行列的值置成特殊值，比如INT_MIN,-1之类的，最后再输出地图时不用管特殊值就可以了**

解题代码：

```cpp
#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,m,k;
    cin>>n>>m>>k;
    vector<vector<int>> grid(n,vector<int>(m));
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            cin>>grid[i][j];
        }
    }
    for(int z=0;z<k;z++){
        int max_num=INT_MIN;
        int row,col;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(grid[i][j]==INT_MIN){
                    continue;
                }
                if(grid[i][j]>max_num){
                    max_num=grid[i][j];//注意这里要给它再重新赋值，要不然无法比较最小值
                    row=i;
                    col=j;
                }
            }
        }
        for(int i=0;i<m;i++){
            grid[row][i]=INT_MIN;
        }
        for(int i=0;i<n;i++){
            grid[i][col]=INT_MIN;
        }
    }
    queue<int> result;//这里直接存入，就不会有那么多的格式问题了
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            if(grid[i][j]!=INT_MIN){
                result.push(grid[i][j]);
            }
            else{
                continue;
            }
        }
    }
   for(int i=0;i<(n-k);i++){
       for(int j=0;j<(m-k);j++){
           if(j>=1){
               cout<<" ";
           }
           cout<<result.front();
           result.pop();
       }
       cout<<endl;
   }
    
    return 0;
}

## L1-096 谁管谁叫爹

题目分析：
本题还是很简单的，理解了题目就行，具体看代码吧

```cpp
#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    long long a,b;
    string A,B;
    for(int i=0;i<n;i++){
        long long Sa=0;
        long long Sb=0; 
        cin>>a>>b;
        A=to_string(a);
        B=to_string(b);
        for(int i=0;i<A.length();i++){
            Sa+=(A[i]-'0');
        }
        for(int i=0;i<B.length();i++){
            Sb+=(B[i]-'0');
        }
        if((a%Sb==0 && b%Sa==0) || (a%Sb!=0 && b%Sa!=0)){
            if(a>b){
                cout<<"A"<<"\n";
            }
            else{
                cout<<"B"<<"\n";
            }
        }
        else if(b%Sa==0 && a%Sb!=0){
            cout<<"B"<<"\n";
        }
        else if(b%Sa!=0 && a%Sb==0){
             cout<<"A"<<"\n";
        }
    }
    return 0;
}

## L1-095 分寝室

题目链接：
https://pintia.cn/problem-sets/994805046380707840/exam/problems/type/7?problemSetProblemId=1649748772841508870

题目分析：本题还是比较简单的，按照题目说的做就行了，简单模拟

```cpp
#include<iostream>
#include<vector>
#include<climits>
using namespace std;
int main(){
    int n0,n1,n;
    cin>>n0>>n1>>n;
    vector<pair<int,int>> solution;
    for(int i=1;i<n;i++){
        int x,y;
        //i是女生寝室数
        if((n0%i)==0 && (n1%(n-i))==0){
            x=n0/i;
            y=n1/(n-i);
            if(x!=1 && y!=1){
                solution.push_back({x,y});
            }
        }
    }
    vector<int> result(2);
    if(solution.empty()){
        cout<<"No Solution";
    }
    else{
        int diff=INT_MAX;
        for(int i=0;i<solution.size();i++){
            int a=solution[i].first;
            int b=solution[i].second;
            if(abs(a-b)<diff){
                diff=abs(a-b);
                result[0]=n0/a;;
                result[1]=n1/b;
            }
        }
        cout<<result[0]<<" "<<result[1];
    }
    
    return 0;
}
